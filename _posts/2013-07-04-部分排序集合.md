### 快速排序
#### 说明
快排最关键的部分在partion部分。时间复杂度是O（N * logN）。
#### 代码
	//非随机快速排序
	void quickSort(int *a, int begin, int end)
	{
		if(begin >= end)
			return;
	
		/*partion begin*/
		int key = a[end];
		int i = begin - 1;
		int j = begin;
		for(int k = 0 ; k < end - begin + 1; k++)
		{
			if(a[j] < key)
			{
				if(j - i > 1)
				{
					int temp = a[j];
					a[j] = a[i + 1];
					a[i + 1] = temp;
				}
				else
					i++;
			}
			j++;
		}
	
		int temp = a[i + 1];
		a[i + 1] = a[end];
		a[end] = temp;
		/*partion end*/
	
		if(i >= begin)
			quickSort(a, begin, i);
		if(i + 2 <= end)
			quickSort(a, i + 2, end);
	}

### 插入排序
#### 说明
插入排序可以在原地排序。时间复杂度是O（N * N）。
#### 代码
	//插入排序
	void insertSort(int *a, int n)
	{
		for(int i = 1; i < n; i++)
		{
			int value = a[i];
			int j = i - 1;
			while(j >= 0 && a[j] > value)
			{
				a[j + 1] = a[j];
				j--;
			}
			a[j + 1] = value;
		}
	}

#### 代码
	//冒泡排序
	void bubbleSort(int *a, int n)
	{
		for(int i = 0; i < n; i++)
		{
			for(int j = 0; j < n - i - 1; j++)
			{
				if(a[j] > a[j + 1])
				{
					int temp = a[j];
					a[j] = a[j + 1];
					a[j + 1] = temp;
				}
			}
		}
	}
